# Day 01 — Basics

# Ansible Task — Install Nginx on the Server

---
- name: Install Nginx on Pathnex server
  hosts: all
  become: yes

  tasks:
    - name: Install nginx
      yum:
        name: nginx
        state: present


# Terraform Task — Create EC2 (t3.micro)

provider "aws" {
  region = "ap-south-2"
}

resource "aws_instance" "PathnexEC2" {
  ami           = "ami-024ebedf48d280810"
  instance_type = "t3.micro"

  tags = {
    Name = "Pathnex-EC2"
  }
}


# Kubernetes Task — Create Nginx Pod

apiVersion: v1
kind: Pod
metadata:
  name: pathnex-pod
spec:
  containers:
    - name: web
      image: nginx:latest
      ports:
        - containerPort: 80

#!/bin/bash

# Git Integration

# Jenkins Pipeline — Checkout Git

pipeline {
    agent any
    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/Pathnex/sample-java-app.git'
            }
        }
        stage('List Files') {
            steps {
                sh 'ls -la'
            }
        }
    }
}

# GitLab CI — Checkout Git

stages:
  - checkout

git-checkout:
  stage: checkout
  script:
    - git clone https://github.com/Pathnex/sample-java-app.git
    - cd sample-java-app
    - ls -la
